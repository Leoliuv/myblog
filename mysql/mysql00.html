# 初始化安装：
bin/mysqld --initialize-insecure --basedir=/paas/iddbs/liuli/mysql01/mysql --datadir=/paas/iddbs/liuli/mysql01/mysql/21000/
# 启动服务：
bin/mysqld_safe --defaults-file=/paas/iddbs/liuli/mysql01/mysql/etc/my21000.cnf &

bin/mysqld_safe --defaults-file=/paas/iddbs/liuli/mysql01/mysql/etc/my22000.cnf --plugin-load="rpl_semi_sync_master=semisync_master.so;rpl_semi_sync_slave=semisync_slave.so" &

# 停止服务:
bin/mysqladmin -umysql -pmysql --socket=/paas/iddbs/liuli/mysql01/mysql/22000/mysql.sock shutdown
# 登入MySQL：
bin/mysql -uroot -proot --socket=/paas/iddbs/liuli/mysql01/mysql/20000/mysql.sock
bin/mysql -uroot -proot -h172.18.231.117 -P21000
# 修改密码：
alter user root@'localhost' identified by 'root';
# 创建用户：
create user mysql@’%' identified by 'mysql';
# 分配权限：
grant all privileges on *.* to mysql@'%' with grant option;
flush privileges;
# 配置主从：
mysql>
show master status;
change master to master_host='127.0.0.1', master_port=21000, master_user='root', master_password='root', master_log_file='log-bin.00005', master_log_pos=2007;
# 开启从线程：
start slave;
show slave\G;
