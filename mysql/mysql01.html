# MySQL安装、初始化、启停操作、添加用户、权限管理

# 1、安装

通过MySQL二进制压缩包进行安装

tar -zxvf 压缩包
# 2、初始化
bin/mysqld --initialize-insecure --basedir=/paas/iddbs/liuli/mysql01/mysql --datadir=/paas/iddbs/liuli/mysql01/mysql/21000/

在初始化时会创建21000文件夹

# 3、启停操作

3.1进行配置操作

mkdir etc
vi my21000.cnf

在配置文件中添加配置参数为：

[mysqld] 
server-id				=11721000	##唯一标识
port 					=21000	##端口号
character-set-server 		=utf8 
basedir				=/paas/iddbs/liuli/mysql01/mysql	##MySQL安装的绝对路径 
datadir 				=/paas/iddbs/liuli/mysql01/mysql/20000/ 	##MySQL数据存放的绝对路径
tmpdir 				=/paas/iddbs/liuli/mysql01/mysql/20000/ 	##存放临时文件的目录
socket 				=/paas/iddbs/liuli/mysql01/mysql/20000/mysql.sock 	##套接字文件
log_error				=/paas/iddbs/liuli/mysql01/mysql/20000/error.log 	##错误日志存放的绝对路径
pid-file 				=/paas/iddbs/liuli/mysql01/mysql/20000/mysql.pid 	##pid的绝对路径，当MySQL启动时，*会将自己的进程ID写入文件中*
log-bin 				=/paas/iddbs/liuli/mysql01/mysql/20000/mysql-bin	##binlog存放的绝对路径

开启服务：

bin/mysqld_safe --defaults-file=/paas/iddbs/liuli/mysql01/mysql/etc/my21000.cnf &
./bin/mysqld_safe --defaults-file=/paas/iddbs/liuli/mysql01/mysql/etc/my21000.cnf &

登入mysql：（第一次登入不需要密码，需要在登入后修改密码）

bin/mysql -uroot --socket=/paas/iddbs/liuli/mysql01/mysql/21000/mysql.sock -uroot

修改密码：

ALTER USER root@'localhost' IDENTIFIED BY 'root';
GRANT ALL PRIVILEGES ON *.* TO 'root'@'localhost' WITH GRANT OPTION;
flush privileges;

停止服务：

bin/mysqladmin -uroot -proot --socket=/paas/iddbs/liuli/mysql01/mysql/21000/mysql.sock shutdown
# 4、添加/删除/修改用户
CREATE USER mysql@'%' IDENTIFIED BY 'mysql';
DROP USER mysql@'%' IDENTIFIED BY 'mysql';
rename user 'username'@'ip地址' to 'new_username'@'new_ip地址'
set password for 'username'@'ip地址' = Password('new_password') 大写P
# 5、增加\删除管理
GRANT ALL PRIVILEGES ON . TO 'mysql'@'%' WITH GRANT OPTION; 
flush privileges;
revoke 权限 on 数据库.表 from 'username'@'ip地址'
flush privileges;

cat *.log | tail -n 行数 从后向前查看日志
